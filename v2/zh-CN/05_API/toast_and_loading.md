##### 用法：

```js
/* 显示一个文字提示，3秒后关闭 */
$toast.show('这是一个提示', 3000).then(() => {
  console.log('toast hide')
})

/* 显示一个文字提示，稍后手动关闭 */
$toast.show('连接中..', 1E10)

/* 手动关闭 */
$toast.hide()

/* 显示一个加载提示，稍后手动关闭 */
$loading.show('拼命加载中..').then(() => {
  console.log('loading showed')

  $loading.hide(2000).then(() => {
    console.log('loading hide')
  })
})
```
